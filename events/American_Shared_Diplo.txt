#Nation asks to join Democratic League
country_event = {
	id = american_diplo.15
	title = american_diplo.15.t
	desc = american_diplo.15.d
	picture = nato_usa_flag
	
	is_triggered_only = yes
	
	option = { #Ok
		name = american_diplo.15.o1
	    ai_chance = {
			factor = 80
		}
		if = {
			limit = { PCA = { is_faction_leader = yes } }
			PCA = {
				country_event = { days = 1 id = american_diplo.16 } 
				add_to_faction = ROOT
			}
		}
		else = {
			CAN = {
				country_event = { days = 1 id = american_diplo.16 } 
				add_to_faction = ROOT
			}
		}
	}
	
	option = { #No
		name = american_diplo.15.o2
	    ai_chance = {
			factor = 20
		}
		add_political_power = -20
		if = {
			limit = { PCA = { is_faction_leader = yes } }
			PCA = {
				country_event = { days = 1 id = american_diplo.17 } 
			}
		}
		else = {
			CAN = {
				country_event = { days = 1 id = american_diplo.17 } 
			}
		}
	}
}

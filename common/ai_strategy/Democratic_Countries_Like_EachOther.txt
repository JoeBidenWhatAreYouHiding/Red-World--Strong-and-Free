democratic_league_membership = {

	allowed = {
		OR = {		
			tag = AST
			tag = HAW
			
		}
	}
	
	enable = {
		CAN = { is_in_faction = yes }
		
		has_democratic_government = yes
		
		CAN = {
			has_democratic_government = yes
		}
	}
	
	abort = {
		OR = {
			has_war_with = CAN
			CAN = { is_in_faction = no }
			
			has_democratic_government = no
			
			CAN = {
				has_democratic_government = no
			}
			
			CAN = { exists = no }
		}
	}
	
	ai_strategy = {
		type = befriend
		id = "CAN"
		value = 400
	}
	
	
	ai_strategy = {
		type = alliance
		id = "CAN"
		value = 120
	}
	
	
}

CAN_likes_democratic_america = {

	allowed = {
		tag = CAN
	}

	enable = {
		has_democratic_government = yes
		is_in_faction = yes

		any_country = { #this should only ever be one country
			has_country_flag = the_united_states_are_back
			has_democratic_government = yes
		}
	}

	abort = {
		OR = {
			has_democratic_government = no
			is_in_faction = no

			}
		}
	}

	#Will improve relations with all American unifiers, but only one should ever exist at a time

		value = 400
	}

		value = 120

#Tanks
RAJ_Heavy_Vehicles_Factory = {
	include = generic_mobile_tank_organization
	icon = GFX_idea_RAJ_Heavy_Vehicles_Factory
	
	allowed = {	
	or = {
		original_tag = RAJ
		original_tag = PUI
		original_tag = RUB
		}
	}

	equipment_type = { mio_cat_eq_all_light_tank mio_cat_eq_all_medium_tank mio_cat_eq_all_heavy_tank }
	research_categories = { mio_cat_tech_light_armor_and_modules mio_cat_tech_medium_armor_and_modules mio_cat_tech_heavy_armor_and_modules }

	initial_trait = {
		organization_modifier = {
			military_industrial_organization_research_bonus = 0.1
		}
		equipment_bonus = {
			maximum_speed = 0.1
		}
	}
	
	available = {
		# When in a Foreign MIO, countries need to be at peace with original country
		IF = {
			limit = {
				FROM = { NOT = { original_tag = RAJ } }
			}
			FROM = { NOT = { has_war_with = RAJ has_war_with = PUI has_war_with = RUB } }
		}
	}
}

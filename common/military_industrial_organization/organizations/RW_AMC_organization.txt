BAE_systems_commonwealth = {
	include = generic_infantry_tank_organization
	icon = GFX_idea_BAE_systems_commonwealth
	
	allowed = {	
		original_tag = AMC
	}

	equipment_type = { mio_cat_eq_all_light_tank mio_cat_eq_all_medium_tank mio_cat_eq_all_heavy_tank }
	research_categories = { mio_cat_tech_light_armor_and_modules mio_cat_tech_medium_armor_and_modules mio_cat_tech_heavy_armor_and_modules }

	initial_trait = {
		organization_modifier = {
			military_industrial_organization_research_bonus = 0.10
		}
		equipment_bonus = { defense = 0.05 soft_attack = 0.05 }
	}
	
	available = {
		# When in a Foreign MIO, countries need to be at peace with original country
		IF = {
			limit = {
				FROM = { NOT = { original_tag = AMC } }
			}
			FROM = { NOT = { has_war_with = AMC } }
		}
	}
}

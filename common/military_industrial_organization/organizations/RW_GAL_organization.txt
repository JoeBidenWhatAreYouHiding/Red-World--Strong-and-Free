#Tanks
urovesa_tank = {
	include = generic_tank_organization	
	icon = GFX_idea_UROVESA_company
	
	allowed = {	
		original_tag = GAL
	}
	
	available = {
		# When in a Foreign MIO, countries need to be at peace with original country
		IF = {
			limit = {
				FROM = { NOT = { original_tag = GAL } }
			}
			FROM = { NOT = { has_war_with = GAL } }
		}
	}
}

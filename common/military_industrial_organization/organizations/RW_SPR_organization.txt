#Tanks
santa_barbara_systems_valencia = {
	include = generic_heavy_tank_organization
	icon = GFX_idea_Santa_Barbara_Systems
	
	allowed = {	
		OR = {
			tag = SPN
			tag = SPR
		}
	}

	available = {
		# When in a Foreign MIO, countries need to be at peace with original country
		IF = {
			limit = {
				FROM = { NOT = { original_tag = SPN } }
			}
			FROM = { NOT = { has_war_with = SPN } }
		}
		IF = {
			limit = {
				FROM = { NOT = { original_tag = SPR } }
			}
			FROM = { NOT = { has_war_with = SPR } }
		}
	}
}

#Tanks
cpim_tanque = {
	include = generic_tank_organization	
	icon = GFX_idea_CPIM
	
	allowed = {	
		original_tag = CEA
	}
	
	available = {
		# When in a Foreign MIO, countries need to be at peace with original country
		IF = {
			limit = {
				FROM = { NOT = { original_tag = CEA } }
			}
			FROM = { NOT = { has_war_with = CEA } }
		}
	}
}

defined_text = {
	name = GetCoalitionPartyPop
	# Recursive Scripted Loc!
	text = { #Coalition present
		trigger = {
			check_variable = { ideology_index_next < global.ideology_groups^num } #End loop if past final index
			check_variable = { coalitions_array^ideology_index_next = 2 } #Checks for presence of coalition
			set_temp_variable = { ideology_index_current = ideology_index_next }
			add_to_temp_variable = { ideology_index_next = 1 }
		}
		localization_key = party_pop_coalition_ideology
	}
	text = { #Coalition not present or is ruling party, move to next ideology
		trigger = {
			check_variable = { ideology_index_next < global.ideology_groups^num } #End loop if past final index
			set_temp_variable = { ideology_index_current = ideology_index_next }
			add_to_temp_variable = { ideology_index_next = 1 }
		}
		localization_key = "[GetCoalitionPartyPop]"
	}
}
defined_text = {
	name = GetCoalitionStability
	text = {
		trigger = {
			if = {
				limit = { NOT = { check_variable = { coalition_stability_array^ideology_index_current = 0 } } }
				set_temp_variable = { coalition_stab = coalition_stability_array^ideology_index_current }
			}
			else = {
				set_temp_variable = { coalition_stab = coalition_stability?0.5 }
			}
		}
		localization_key = party_pop_coalition_stability
	}
}
defined_text = {
	name = GetCoalitionPPGain
	text = {
		trigger = {
			set_temp_variable = { coalition_partner = global.ideology_groups^ideology_index_current }
			set_temp_variable = { coalition_pp = party_popularity@var:coalition_partner }
			if = {
				limit = { NOT = { check_variable = { coalition_stability_array^ideology_index_current = 0 } } }
				multiply_temp_variable = { coalition_pp = coalition_stability_array^ideology_index_current }
			}
			else = {
				multiply_temp_variable = { coalition_pp = coalition_stability?0.5 }
			}
		}
		localization_key = party_pop_coalition_pp_gain
	}
}

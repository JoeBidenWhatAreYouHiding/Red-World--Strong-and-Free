APC_establish_syndicates = {

	NEA_establish_syndicate = {

		name = NEA_establish_a_syndicate

		icon = establish_syndicate

		allowed = {
			original_tag = NEA
		}

		available = {
			FROM = {
				is_controlled_by = NEA
				OR = {
					is_island_state = yes
					any_neighbor_state = {
						is_controlled_by = ROOT
					}
				}
			}
		}

		state_target = yes

		on_map_mode = map_only

		#targets = {  }

		target_array = controlled_states

		target_trigger = {
			has_completed_focus = NEA_establish_syndicates
			FROM = {
				NOT = {
					is_core_of = ROOT
				}
			}
		}

		cost = 50

		fire_only_once = yes

		days_remove = 50

		modifier = {
			political_power_gain = -0.5
		}

		visible = {
			tag = NEA
			has_completed_focus = NEA_establish_syndicates
			always = yes
		}

		complete_effect = {

		}

		remove_effect = {
			FROM = {
				add_compliance = 10
				add_resistance = -10
				add_state_modifier = {
    				modifier = {
        				compliance_gain = 0.1
    				}
				}
			}
		}
	}
}
